{

     VAR 
        INTEGER :: I1 // I2 // TabI[10] $ 
        REAL :: R1 // R2 // TabR[20] $

    CONST

    /* ------ Partie instructions ------ */

    /* ------ Affectation -------- */

    READ( " % "  | @ R1 ) $
    
    WRITE("la valeur de idf %" | R2) $

    while( I1 INF 10)
    {
        R1 := ( 1 + ( 0 / 4 ) ) $

        while(I2 EG TabI[3])
        {
            R2 := R2 + 1.5 $
            /* ----- la condition IF EXECUT ELSE END_IF ----- */

                I2 := I1 + 23 $
            IF( 2 INF 5 ) ELSE 
            EXECUT
                I2 := 23 + 23 $
            END_IF $
        }
    }

    /* ----- la condition IF EXECUT END_IF ----- */

        I2 := 23 + 23 $
    IF( 2 INF 5 )
    END_IF $
}
